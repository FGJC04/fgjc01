static int ReverseInteger(int num)
{
   int reversed = 0;
   while (num != 0)
     {
     int remainder = num % 10;
      reversed = reversed * 10 + remainder;
      num /= 10;
     }
    return reversed;
}
int num;
do{
  Console.Write($"Enter a three digit number :");
  num = int.Parse(Console.ReadLine());
} while (num < 100 || num > 999);
Console.ForegroundColor = ConsoleColor.Green;
int rev = ReverseInteger(num);
Console.WriteLine(rev);
int minus = (num > rev) ? num - rev : rev - num;
int arev = ReverseInteger(minus);
Console.WriteLine(arev);
int equal = minus + arev;
Console.WriteLine(equal);
string sen1 = (num > rev) ?
$"the entered value : {num} - reverse of the value : {rev} equals {minus}":
$"the entered value : {rev} - reverse of the value : {num} equals {minus}";
Console.WriteLine(sen1);
string sen2 = $"the minus value : {minus} + the reverse of minus value : {arev} = {equal}";
Console.WriteLine(sen2);
Console.WriteLine("Note : every three digit that would minus to its reverse(depent on the highest)" +
$" \n then the defference of previus calculation plus the reverse \n will always equal to : {equal} ");
